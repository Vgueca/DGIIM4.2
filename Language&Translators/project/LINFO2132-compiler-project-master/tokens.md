# Tokens

- separator: ' ', '\t', '\n'
- comment: '//'
- identifier: 'a-zA-Z0-9\_'
- keywords: const, record, var, val, proc, for, to, by, while, if, else, return, and, or

- int: 0-9+
- real: 0-9+.
- string: "..."
- bool: true | false

- assignment: =
- addition: +
- subtraction: -
- multiplication: \*
- division: /
- modulo: %
- equal: ==
- different: <>
- smaller: <
- greater: >
- seq: <=
- geq: >=
- parenthesis: ( | )
- curlybrackets: { | }
- brackets: [ | ]
- dot: .
- comma: ,
- semicolon: ;
